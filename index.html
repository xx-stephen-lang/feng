        function createScoreSelector(dimension, index) {

            const container = document.createElement('div');

            container.className = 'score-selector';

            

            dimension.criteria.forEach((criterion, i) => {

                const score = i + 1;

                const option = document.createElement('div');

                option.className = 'score-option';

                option.innerHTML = `

                    <input type="radio" name="score-${index}" value="${score}" 

                        id="score-${index}-${score}" ${score === 1 ? 'checked' : ''}>

                    <label for="score-${index}-${score}">

                        <span class="score-value">${score}分</span>

                        <span class="score-desc">${criterion}</span>

                    </label>

                `;

                container.appendChild(option);

            });


            return container;

        }


        function loadDimensions() {

            const dimensions = model.dimensions;

            let html = `<div class="dimensions-grid">`;

            

            dimensions.forEach((dim, index) => {

                html += `

                    <div class="dimension-card">

                        <div class="dimension-header">

                            <h3>${dim.name}</h3>

                        </div>

                        <div class="score-container" id="score-container-${index}"></div>

                    </div>`;

            });


            html += `</div>`;


            document.getElementById("dimensions-content").innerHTML = html;


            dimensions.forEach((dim, index) => {

                const container = document.getElementById(`score-container-${index}`);

                container.appendChild(createScoreSelector(dim, index));

            });


            // 监听评分变化，启用按钮

            const inputs = document.querySelectorAll('input[type="radio"]');

            inputs.forEach(input => {

                input.addEventListener('change', () => {

                    const calculateBtn = document.getElementById('calculate-btn');

                    calculateBtn.disabled = false;

                    calculateBtn.textContent = '生成评估结果';

                });

            });

        }


        function calculateScore() {

            const dimensions = model.dimensions;

            let judicialScore = 0, prScore = 0;


            dimensions.forEach((dim, index) => {

                const selectedScore = parseInt(

                    document.querySelector(`input[name="score-${index}"]:checked`).value

                );

                judicialScore += selectedScore * dim.weightLegal / 100;

                prScore += selectedScore * dim.weightPR / 100;

            });
